#Create Comments table 
CREATE TABLE IF NOT EXISTS `Comments` (
  `CommentID` int UNSIGNED NOT NULL AUTO_INCREMENT,

  `UserComment` VARCHAR(100),
  `TimeStamp` TimeStamp,
  `Date` Date,
  `has_comment` int,
  PRIMARY KEY (CommentID)
) ENGINE = InnoDB;
